Your algorithm for TD prediction has five arguments:

  • V: This is a dictionary where V[s] is the estimated value of state s.

def td_prediction(env, num_episodes, policy, alpha, gamma=1.0):
    '''
    Creates a value dictionary of all the board spaces the agent encounters.
    '''
    # initialize empty dictionaries of floats
    V = defaultdict(float)
    # loop over episodes
    for i_episode in range(1, num_episodes+1):
        # monitor progress
        if i_episode % 100 == 0:
            print("\rEpisode {}/{}".format(i_episode, num_episodes), end="")
            sys.stdout.flush()
        
        ## TODONE: complete the function
        
        # Observe State 0
        state = env.reset()
        
        while True:
            
            # Choose Action @ t using policy
            action = policy[state]
            
            # Take Action @ t and observe R[t+1] and S[t+1]
            next_state, reward, done, info = env.step(action)
            
            # TD(0) formula
            V[state] = V[state] + alpha * (reward + gamma * V[next_state] - V[state])
        
            state = next_state
            
            # Check for terminal state
            if done:
                break
        
    return V

def choose_action(env, Q, epsilon, state):
    '''
    Uses an epsilon greedy formula to choose the action
    '''
    nA = env.nA
    # Checks if state has already come up; if so, creates probabilities using the epsilon greedy method
    if state in Q:
        probs = np.ones(nA) * (epsilon / nA)
        best_a = np.argmax(Q[state])
        probs[best_a] = 1 - epsilon + (epsilon / nA)

    # If it's a new state, it uses even probability.
    else:
        probs = np.full(nA, 1 / nA)

    # Choose A[0] using e-greedy policy        
    action = np.random.choice(np.arange(nA), p=probs)
    
    return action

def sarsa(env, num_episodes, alpha, gamma=1.0):
    '''
    Using the Sarsa(0) method, creates an action-value function without a stated policy
    '''
    nA = env.nA
    # initialize action-value function (empty dictionary of arrays)
    Q = defaultdict(lambda: np.zeros(env.nA))
    # initialize performance monitor
    # loop over episodes
    for i_episode in range(1, num_episodes+1):
        # monitor progress
        if i_episode % 100 == 0:
            print("\rEpisode {}/{}".format(i_episode, num_episodes), end="")
            sys.stdout.flush()   
        
        ## TODONE: complete the function
        
        # Set epsilon to scale with the number of episodes
        epsilon = 1. / i_episode
        
        # Observe State 0
        state = env.reset()

        while True:
            # Set Action[t]
            action = choose_action(env, Q, epsilon, state)
            
            # Take Action @ t and observe R[t+1] and S[t+1]
            next_state, reward, done, info = env.step(action)
            
            # Set Action[t+1]
            next_action = choose_action(env, Q, epsilon, next_state)
            
            # Sarsa(0) formula
            Q[state][action] = Q[state][action] + alpha * (reward + gamma * Q[next_state][next_action] - Q[state][action])
        
            state = next_state
            
            # Check for terminal state
            if done:
                break
        
    return Q

def q_learning(env, num_episodes, alpha, gamma=1.0):
    '''
    Using the Sarsamax/Q-learning method, creates an action-value function without a stated policy
    '''
    # initialize empty dictionary of arrays
    Q = defaultdict(lambda: np.zeros(env.nA))
    # loop over episodes
    for i_episode in range(1, num_episodes+1):
        # monitor progress
        if i_episode % 100 == 0:
            print("\rEpisode {}/{}".format(i_episode, num_episodes), end="")
            sys.stdout.flush()
        
        ## TODONE: complete the function
        
        # Set epsilon to scale with the number of episodes
        epsilon = 1. / i_episode
        
        # Observe State 0
        state = env.reset()

        while True:
            # Set Action[t]
            action = choose_action(env, Q, epsilon, state)
            
            # Take Action @ t and observe R[t+1] and S[t+1]
            next_state, reward, done, info = env.step(action)
            
            # Sarsamax formula
            Q[state][action] = Q[state][action] + alpha * (reward + gamma * max(Q[next_state]) - Q[state][action])
        
            state = next_state
            
            # Check for terminal state
            if done:
                break
        
    return Q

def expected_sarsa(env, num_episodes, alpha, gamma=1.0):
    # initialize empty dictionary of arrays
    nA = env.nA
    Q = defaultdict(lambda: np.zeros(nA))
    # loop over episodes
    for i_episode in range(1, num_episodes+1):
        # monitor progress
        if i_episode % 100 == 0:
            print("\rEpisode {}/{}".format(i_episode, num_episodes), end="")
            sys.stdout.flush()
        
        ## TODONE: complete the function
        
        # Set epsilon to a constant, since it doesn't work if it scales with the number of episodes
        epsilon = 0.0005 # 1. / i_episode
        
        # Observe State 0
        state = env.reset()

        while True:
            # Set Action[t]
            action = choose_action(env, Q, epsilon, state)
            
            # Take Action @ t and observe R[t+1] and S[t+1]
            next_state, reward, done, info = env.step(action)
            
            
            # Expected Sarsa formula
            probs = np.ones(nA) * (epsilon / nA)
            probs[np.argmax(Q[next_state])] = 1 - epsilon + (epsilon / nA)
            
            expected = np.dot(Q[next_state], probs)
            
            Q[state][action] = Q[state][action] + (alpha * (reward + (gamma * expected) - Q[state][action]))
        
            state = next_state
            
            # Check for terminal state
            if done:
                break
        
    return Q
